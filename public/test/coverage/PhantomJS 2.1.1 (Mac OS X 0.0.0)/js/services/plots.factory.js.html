(function() {
  'use strict';
  
  angular
    .module('MedicionAguaAdmin')
    .factory('PlotsService', factory);
  
  factory.$inject = ['$q','$http','APP_SETTINGS'];
  
  /* @ngInject */
  function factory($q, $http, APP_SETTINGS) {
    
    var service = {
      getCondos: getCondos,
      getPlots: getAll
    };
    return service;
    
    //Generic functions
    function resolvePromise(result){
      return $q.when(result.data);
    }
    function rejectPromise(err){
      return $q.reject(err);
    }
 
    //Service implementations
    function getCondos() {
      return $http
              .get(APP_SETTINGS.BASE_URL + '/condos')
              .then(resolvePromise, rejectPromise);
    }
 
    function getAll(_condoId) {
 
      var query = {
        params: {
          condoId : _condoId
        }
      };
 
      return $http
              .get(APP_SETTINGS.BASE_URL + '/plots')
              .then(resolvePromise, rejectPromise);
    }
 
 
  }
})();
