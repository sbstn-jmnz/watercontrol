(function() {
  'use strict';
  
  angular
    .module('MedicionAguaAdmin')
    .controller('PlotsController', Controller);
  
  Controller.$inject = ['$rootScope', '$filter','$timeout','PlotsResolve', 'CondosResolve'];
  
  /* @ngInject */
  function Controller($rootScope, $filter, $timeout, PlotsResolve, CondosResolve) {
    
    var vm = this;
    //Variables
    vm.condosList = angular.copy(CondosResolve);
    vm.plotsList = angular.copy(PlotsResolve);
    vm.searchModel = '';
    vm.selectedCondo = vm.condosList[0].id;
    
    //Funciones
    vm.searchPlot = search;
 
 
    //CARGAR PLOTS CON SPINNER
    $rootScope.$broadcast('open:spinner');
    $timeout(function(){
      $rootScope.$broadcast('close:spinner');
    }, 1000);
 
    //Implementaciones
    function search(_searchModel){
 
      vm.plotsList = _searchModel? $filter('filter')(vm.plotsList, {plot_number: _searchModel})
                                 : angular.copy(PlotsResolve);
    }
 
  }
})();
